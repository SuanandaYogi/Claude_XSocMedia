import React, { useState, useEffect } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Bell, Brain, DollarSign, Heart, Clock, Users, AlertTriangle } from 'lucide-react';

export default function DigitalExploitation() {
  const [notifications, setNotifications] = useState(0);
  const [mentalHealth, setMentalHealth] = useState(100);
  const [profit, setProfit] = useState(0);
  const [timeSpent, setTimeSpent] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      setNotifications(prev => prev + 1);
      setMentalHealth(prev => Math.max(0, prev - 0.5));
      setProfit(prev => prev + 2);
      setTimeSpent(prev => prev + 1);
    }, 1000);

    return () => clearInterval(timer);
  }, []);

  return (
    <div className="p-4 max-w-4xl mx-auto space-y-6">
      <Card className="bg-white shadow-lg">
        <CardHeader>
          <CardTitle className="text-2xl text-center text-red-600 flex items-center justify-center gap-2">
            <AlertTriangle className="w-8 h-8" />
            Digital Exploitation Reality Check
          </CardTitle>
        </CardHeader>
        <CardContent>
          <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
            {/* Notifications Counter */}
            <Card className="p-4 bg-blue-50">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-2">
                  <Bell className="w-6 h-6 text-blue-600" />
                  <h3 className="font-semibold">Triggers Sent</h3>
                </div>
                <span className="text-xl font-bold text-blue-600">{notifications}</span>
              </div>
              <p className="mt-2 text-sm text-gray-600">
                Every second, platforms send new triggers to recapture your attention
              </p>
            </Card>

            {/* Mental Health Impact */}
            <Card className="p-4 bg-red-50">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-2">
                  <Brain className="w-6 h-6 text-red-600" />
                  <h3 className="font-semibold">Mental Wellbeing</h3>
                </div>
                <span className="text-xl font-bold text-red-600">{mentalHealth.toFixed(1)}%</span>
              </div>
              <div className="mt-2 w-full bg-gray-200 rounded-full h-2.5">
                <div 
                  className="bg-red-600 h-2.5 rounded-full transition-all duration-500"
                  style={{ width: `${mentalHealth}%` }}
                ></div>
              </div>
            </Card>

            {/* Corporate Profit */}
            <Card className="p-4 bg-green-50">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-2">
                  <DollarSign className="w-6 h-6 text-green-600" />
                  <h3 className="font-semibold">Corporate Profit</h3>
                </div>
                <span className="text-xl font-bold text-green-600">${profit}</span>
              </div>
              <p className="mt-2 text-sm text-gray-600">
                Companies profit from every moment of engagement
              </p>
            </Card>

            {/* Time Spent */}
            <Card className="p-4 bg-purple-50">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-2">
                  <Clock className="w-6 h-6 text-purple-600" />
                  <h3 className="font-semibold">Time Lost</h3>
                </div>
                <span className="text-xl font-bold text-purple-600">{timeSpent}s</span>
              </div>
              <p className="mt-2 text-sm text-gray-600">
                Every second spent is engineered for addiction
              </p>
            </Card>
          </div>

          {/* Key Messages */}
          <div className="mt-6 space-y-4">
            <div className="p-4 bg-yellow-50 rounded-lg">
              <h3 className="font-bold text-lg text-yellow-800 mb-2">Remember:</h3>
              <ul className="space-y-2 text-yellow-700">
                <li className="flex items-center gap-2">
                  <AlertTriangle className="w-4 h-4" />
                  "Healthy usage" of platforms designed for addiction is a myth
                </li>
                <li className="flex items-center gap-2">
                  <Brain className="w-4 h-4" />
                  Systems are designed to exploit human psychology
                </li>
                <li className="flex items-center gap-2">
                  <Heart className="w-4 h-4" />
                  Your wellbeing matters more than digital engagement
                </li>
                <li className="flex items-center gap-2">
                  <Users className="w-4 h-4" />
                  Real connections trump digital interactions
                </li>
              </ul>
            </div>
          </div>

          <div className="mt-6 text-center text-sm text-gray-600">
            This visualization demonstrates how digital platforms continuously work to capture and monetize your attention, regardless of the human cost.
          </div>
        </CardContent>
      </Card>
    </div>
  );
}
